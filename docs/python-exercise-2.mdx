### Task 3: Create Vector Search Index
```python
VS_INDEX_NAME = "vector_index_voyageai"

# Define vector index configuration
model = {
    "name": VS_INDEX_NAME,
    "type": "vectorSearch",
    "definition": {
        "fields": [
            {
                "type": "vector",
                "path": "embedding",
                "numDimensions": 1024,
                "similarity": "cosine",
            }
        ]
    },
}

# Create the vector search index
collection.create_search_index(model=model)
print(f"ğŸ¯ Vector search index '{VS_INDEX_NAME}' created!")

# Check index status
indexes = list(collection.list_search_indexes())
for idx in indexes:
    print(f"ğŸ” Index: {idx.get('name')} - Status: {idx.get('status')}")
```

```python
def get_information_for_question_answering(user_query: str):

            }
        },
        {
            "$project": {
                "_id": 0,
                "key": 1,
                "page_number": 1,
                "score": {"$meta": "vectorSearchScore"},
            }
        },
    ]
    
    # Execute search
```
